public class Task58 extends TaskMain{
public static void main(String[] args) {}
	
	
	
	public static void strCount(String str, char letter) {
	    //write code here
	    int count = 0;
	    char[] chr = str.toCharArray();
	    for(int i = 0; i != chr.length; i++){
	      if(chr[i] == letter) count += 1;
	    }
	    System.out.println(count);
	  }
}
